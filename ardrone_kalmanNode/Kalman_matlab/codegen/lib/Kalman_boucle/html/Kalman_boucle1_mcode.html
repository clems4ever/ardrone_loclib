1function [ Xk,Pk,Prediction ] = Kalman_boucle(Cmesure, R, Xprec1, Pprec1, Zprec1 )
 2%KALMAN_boucle Summary of this function goes here
 3%   Detailed explanation goes here
 4% Boucle de filtre de Kalman
 5    
 6%Initialisations
 7    %Temps d'échantillonnage
 8    Te = 1;
 9    %Process noise
10    incertitude_modele = 1;
11    Q = incertitude_modele * eye(2);
12    M = 1;
13    
14    Xk=zeros(2,1);  %6 si on prend en compte l'accélération
15    Prediction=zeros(2,1);  %6 si on prend en compte l'accélération
16    Pk=eye(2);     %6 si on prend en compte l'accélération
17    Zk=zeros(8,1); %10 si on prend en compte l'accélération
18    In=zeros(8,1); %10
19    Xprec=zeros(2,1);
20    Xprec=Xprec1;
21    C=zeros(8,2);
22    C=Cmesure; 
23    
24    Pprec=zeros(2,2);
25    Zprec=zeros(8,1);
26    Pprec=Pprec1;
27    Zprec=Zprec1;
28    
29    % Matrice d'état
30    A = [1  0       %position, vitesse sur X
31         0  1];      %position, vitesse sur Y 
32         
33
34    % Matrice des mesures
35%     C=[ 1 0      %position X     GPS
36%         0 1      %position Y     GPS
37%         1 0      %position X     ODOM
38%         0 1      %position Y     ODOM
39%         1 0      %position X     TAG
40%         0 1      %position Y     TAG
41%         1 0      %position X     XTUM
42%         0 1];    %position Y     YTUM
43
44
45    
46    
47    
48    %Prediction
49    Xk=A*Xprec;
50    Pk=A*Pprec+(M*Q*M');
51        
52    %Innovation
53    In=Zprec-(C*Xk);
54    Kk=Pk*C'*inv((C*Pk*C')+R);
55    
56    %Mise a jour
57    Xk=Xk+Kk*In;
58    Pk=(eye(2)-Kk*C)*Pk;
59    
60    %Incrementation
61    Prediction(1)=Xk(1); %X
62    Prediction(2)=Xk(2); %Y
63    
64    
65end
